192 void
193 UnscentedKalmanFilter<MotionModelTp,
194 ObservationModelTp>::initialize_sigma_points(real_t k){
195 
198 
199  if(!motion_model_ptr_){
200  throw std::logic_error("Motion model has not been set");
201  }
202 
203  set_k(k);
204 
205  const uint_t n = motion_model_ptr_->get_state().dimension;
206 
207  sigma_points_.resize(2*n + 1, DynVec<real_t>());
208  w_.resize(2*n + 1);
209 
210  w_[0] = k_/(n + k_);
211  auto begin = w_.begin();
212  begin++;
213  std::for_each(begin, w_.end(),
214  [this, n](real_t& w){w = 0.5*(1/n + this->k_);});
215 
217  update_sigma_points();
218 }

221  typename ObservationModelTp>
222 void
223 UnscentedKalmanFilter<MotionModelTp,
224 ObservationModelTp>::update_sigma_points(){
225 
226  auto state_vec = motion_model_ptr_->get_state().as_vector();
227 
228  const uint_t n = state_vec.size();
229 
232  sigma_points_[0] = state_vec;
233 
234  auto& P = (*this)["P"];
237  DynMat<real_t> L;
238  llh(P, L);
239 
242  for(uint_t i=1; i<=n; ++i){
243  auto col1 = column(L, i-1);
244  sigma_points_[i] = state_vec + std::sqrt(n + k_)*col1;
245  sigma_points_[i + n] = state_vec - std::sqrt(n+k_)*col1;
246 
247  }

313 template<typename MotionModelTp, typename ObservationModelTp>
314 void
315 UnscentedKalmanFilter<MotionModelTp,
316  ObservationModelTp>::predict(const motion_model_input_t& u){
317 
319  check_sanity_();
320 
321  static auto point_propagator = [this, &u](DynVec<real_t>& point){
322  point = motion_model_ptr_->evaluate(point, u);
323  };
324 
326  std::for_each(sigma_points_.begin(),
327  sigma_points_.end(),
328  point_propagator);
329 
330  DynVec<real_t> total_state(get_state().dimension, 0.0);
331 
333  for(uint_t p=0; p<sigma_points_.size(); ++p){
334  total_state += w_[p]*sigma_points_[p];
335  }
336 
338  get_state().set(total_state);
339 
340  DynMat<real_t> total_cov(get_state().dimension,
341  get_state().dimension, 0.0);
342 
345  for(uint_t p=0; p<sigma_points_.size(); ++p){
346  total_cov += w_[p]*(sigma_points_[p] - total_state)*trans(sigma_points_[p] - total_state);
347  }
348 
349  auto& P = (*this)["P"];
350  auto& Q = (*this)["Q"];
351 
352  P = total_cov + Q;
353 }

355 template<typename MotionModelTp, typename ObservationModelTp>
356 void
357 UnscentedKalmanFilter<MotionModelTp,
358  ObservationModelTp>::update(const observation_model_input_t& z){
359 
360  check_sanity_();
361 
364  auto zpred_hat = observation_model_ptr_->evaluate(sigma_points_);
365 
368  DynVec<real_t> zpred(zpred_hat[0].size(), 0.0);
369 
370  for(uint_t p = 0; p<sigma_points_.size(); ++p){
371  zpred += w_[p]*zpred_hat[p];
372  }
373 
374  auto& R = (*this)["R"];
375  DynMat<real_t> Py = R;
376 
377  for(uint_t p=0; p<sigma_points_.size(); ++p){
378  Py += w_[p]*(zpred_hat[p] - zpred)*trans(zpred_hat[p] - zpred);
379  }
380 
381  auto Py_invs = inv(Py);
382  auto state_vec = motion_model_ptr_->get_state().as_vector();
383  DynMat<real_t> Pxy(motion_model_ptr_->get_state().dimension,
384  zpred.size(), 0.0);
385 
386  for(uint_t p=0; p<sigma_points_.size(); ++p){
387  Pxy += w_[p]*(sigma_points_[p] - state_vec)*trans(zpred_hat[p] - zpred);
388  }
389 
390  if(has_matrix("K")){
391  auto& K = (*this)["K"];
392  K = Pxy*Py_invs;
393  }
394  else{
395  auto K = Pxy*Py_invs;
396  set_matrix("K", K);
397  }
398 
399  auto& K = (*this)["K"];
400 
402  auto innovation = z - zpred;
403  motion_model_ptr_->get_state() += K*innovation;
404 
405  auto& P = (*this)["P"];
406  P -= K*Py*trans(K);
407 }
408 
409 }
