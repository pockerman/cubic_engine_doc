83  //node is obviously zero at the start node
84  start.data.gcost = 0.0;
85

99  break;
100  }
101 
102  //current node is not the goal so proceed
103  //add it to the explored (or else called closed) set
104  explored.insert(cv);
105 
106  //get the adjacent neighbors
107  std::pair<adjacency_iterator,adjacency_iterator> neighbors = g.get_vertex_neighbors(cv);
108  auto itr = neighbors.first;
109 
111  for(; itr != neighbors.second; itr++){

114 
115  if(open.contains(nv)){
116  continue;
117  }
118  else{
119 
120  // we cannot move to the neighbor
121  // so no reason checking
122  if(!nv.data.can_move()){
123  explored.insert(nv);
124  continue;
125  }
126  }
127 
128 
129  uint_t nid = nv.id;
130 
131  //search explored set by id
132  auto itr = std::find_if(explored.begin(), explored.end(),
133  [=](const node_t& n){return (n.id == nid);});
134 
135  //the node has been explored
136  if(itr != explored.end()){
137  continue;
138  }
139 
140 
141  //this actually the cost of the path from the current node
142  //to reach its neighbor
143  cost_t tg_cost = cv.data.gcost + h(cv.data.position, nv.data.position);
144 
145  if (tg_cost >= nv.data.gcost) {
146  continue; //this is not a better path
147  }
148 
149  // This path is the best until now. Record it!
150  kernel::add_or_update_map(came_from,nv.id,cv.id);
151

153  nv.data.gcost = tg_cost;
154

157

159  open.push(nv);
160 
161  }
