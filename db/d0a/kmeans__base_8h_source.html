206 kmeans_base::detect_convergence(const Similarity& sim,
207  const std::vector<kmeans_detail::Cluster<CentroidType> >& clusters,
208  const std::vector<CentroidType>& old_centers){
209 
210 #ifdef PARML_DEBUG
211  size_type nclusters = clusters.size();
212  size_type nold_centers = old_centers.size();
213  using utilities::ExeSizeMismatch;
214  const std::string msg = "Current number of clusters: "+std::to_string(nclusters)+
215  " not equal of old number of clusters: "+std::to_string(nold_centers);
216  Assert(nclusters == nold_centers,ExeSizeMismatch(msg));
217 #endif
218 
219  bool converged = true;
220 
221  for(size_type c=0; c<clusters.size(); ++c){
222 
223  real_type dis = sim(clusters[c].location,old_centers[c]);
224 
225  if(dis > data_.tolerance){
226  converged = false;
227  break; //we don't have to continue calculating distances
228  }
229  }
230 
231  return converged;
232 }
