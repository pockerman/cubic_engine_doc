256 void
257 KalmanFilter<MotionModelTp,
258  ObservationModelTp>::predict(const DynVec<real_t>& u,
259  const DynVec<real_t>& w){
260 
261  if(!motion_model_ptr_){
262  throw std::runtime_error("Motion model has not been set");
263  }
264 
267  auto& state = motion_model_ptr_->get_state();
268  auto x = state.as_vector();
269 
272  auto& F = motion_model_ptr_->get_matrix("F");
273  auto& B = (*this)["B"];
274 
275  x = F*x + B*u + w;
276  state.set(x);
277 
279  auto& P = (*this)["P"];
280  auto& Q = (*this)["Q"];
281  auto F_T = trans( F );
282 
283  P = (F*P*F_T) + Q;
284 }

287 void
288 KalmanFilter<MotionModelTp,
289  ObservationModelTp>::update(const observation_model_input_t& z){
290 
291  if(!motion_model_ptr_){
292  throw std::runtime_error("Motion model has not been set");
293  }
294 
295  if(!observation_model_ptr_){
296  throw std::runtime_error("Observation model has not been set");
297  }
298 
299  auto& state = motion_model_ptr_->get_state();
300  auto x = state.as_vector();
301  auto& P = (*this)["P"];
302  auto& R = (*this)["R"];
303 
304  auto& H = observation_model_ptr_->get_matrix("H");
305  auto H_T = trans(H);
306 
307  try{
308 
309  auto S = H*P*H_T + R;
310  auto S_inv = inv(S);
311 
312  if(has_matrix("K")){
313  auto& K = matrices_["K"];
314  K = P*H_T*S_inv;
315  }
316  else{
317  auto K = P*H_T*S_inv;
318  set_matrix("K", K);
319  }
320 
321  auto& K = (*this)["K"];
322  auto innovation = z - H*x;
323 
324  if(K.columns() != innovation.size()){
325  throw std::runtime_error("Matrix columns: "+
326  std::to_string(K.columns())+
327  " not equal to vector size: "+
328  std::to_string(innovation.size()));
329  }
330 
331  x += K*innovation;
332  state.set(x);
333 
334  IdentityMatrix<real_t> I(state.size());
335 
337  P = (I - K*H)*P;
338  }
339  catch(...){
340 
341  // this is a singular matrix what
342  // should we do? Simply use the predicted
343  // values and log the fact that there was a singular matrix
344 
345  throw;
346  }
