198 void
199 RelaxedFVMatrixAssemblyPolicy<MatrixPolicy>::relax(const MeshTp& mesh, const vector_t& old_solution){
200 
201  std::vector<real_t> sum_off(D_.size(), 0.0);
202 
203  sum_mag_off_diag(sum_off);
204 
205  std::vector<real_t> internal_coeffs;
206 
207  std::vector<DoF> dofs;
208 
209  ConstElementMeshIterator<Active, MeshTp> filter(mesh);
210 
211  auto begin = filter.begin();
212  auto end = filter.end();
213 
214  for(; begin != end; ++begin){
215 
216  auto* cell = *begin;
217 
218  cell->get_dofs(var_name_, dofs);
219  real_t diag_0 = D_.get(dofs[0].id);
220 
221  for(uint_t f=0; f<cell->n_faces(); ++f){
222 
223  auto& face = cell->get_face(f);
224 
225  if(face.on_boundary()){
226 
227  get_internal_coefficients(face,internal_coeffs);
228 
229  auto min_elem = std::min_element(internal_coeffs.begin(), internal_coeffs.end());
230  D_.add(dofs[0].id, (std::fabs(*min_elem) - *min_elem));
231  sum_off[dofs[0].id] += (std::fabs(*min_elem) - *min_elem);
232  }
233  }
234 
235  //make sure that the matrix remains diagonally dominant
236  D_.set(dofs[0].id,std::max(std::fabs(D_.get(dofs[0].id)),sum_off[dofs[0].id]));
237 
238  //then relax
239  D_.set(dofs[0].id, D_.get(dofs[0].id)/relax_factor_);
240 
241  const real_t old_sol = old_solution.get(dofs[0].id);
242  rhs_.add(dofs[0].id, (D_.get(dofs[0].id) - diag_0)*old_sol);
243  }
244 
245 }
