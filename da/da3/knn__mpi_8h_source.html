71 std::pair<typename Actor::return_type,KnnInfo>
72 Knn<knn_type::Type::MPI_KNN>::predict(const utilities::DataInput<DataMat,DataVec>& data,
73  const DataVec& point,
74  const Similarity& sim,Actor& actor){
75 
76  std::chrono::time_point<std::chrono::system_clock> start, end;
77  start = std::chrono::system_clock::now();
78 
79  size_type k = actor.n_neighbors();
80  size_type nrows = data.t1.n_rows();
81  size_type labels_size = data.t2.size();
82 
83 #ifdef PARML_DEBUG
84  //using more neighbors than the actual rows of the input matrix is an error
85  using utilities::ExeLogicError;
86  const std::string in_msg = "In Knn<knn_type::Type::MPI_KNN>::predict(). ";
87  std::string msg = in_msg + "Number of neighbors: "+std::to_string(k)+
88  " not in [1,"+std::to_string(nrows)+")";
89  Assert(((k>=1) && (k < nrows)),ExeLogicError(msg));
90 
91  msg = in_msg + "Data point is empty.";
92  Assert(point.size() != static_cast<size_type>(0),ExeLogicError(msg));
93 
94  //labels should not be empty and should be of same size
95  //as the number number of training examples
96  msg = in_msg + "Labels size: "+
97  std::to_string(labels_size)+" not equal to: "+std::to_string(nrows);
98 
99  Assert(labels_size == nrows,ExeLogicError(msg));
100 
101  msg = in_msg+" Parallel object does not have partitions";
102  Assert(this->has_partitions(),ExeLogicError(msg));
103 #endif
104 
105 
106  KnnInfo info;
107  info.n_neighbors = k;
108  info.n_pts_predicted = 1;
109  info.nprocs = parframepp::rt_base<mpi_par>::n_processors();
110  info.nthreads = parframepp::rt_base<mpi_par>::n_threads();
111  info.type = knn_type::Type::MPI_KNN;
112 
113  size_type pid = parframepp::rt_base<mpi_par>::pid();
114  size_type master_pid = parframepp::rt_base<mpi_par>::master_process_id();
115 
116  //the processor range
117  utilities::range1d<size_type> proc_range;
118 
119  if(data_dist_type_flag_ == parframepp::data_dist_type::Type::CHUNK_ON_PROC){
120 
121  proc_range = this->get_local_indices(pid);
122  }
123  else if(data_dist_type_flag_ == parframepp::data_dist_type::Type::COPY_ON_PROC){
124 
125  proc_range = this->get_partition(pid);
126  }
127 
128  //make the local computations for the actor
129  actor(data,point,sim,proc_range);
130 
131  //now we have computed the distances. If we do classification
132  //we must send the top-k class indices. if we do regression we must
133  //send the top-k values
134 
135  typedef typename Actor::comm_chunk comm_chunk;
136  boost::scoped_array<comm_chunk> proc_results;
137 
138  //get the result computed on the processor
139  actor.get_top_k_results(proc_results);
140 
141  boost::scoped_array<comm_chunk> results;
142 
143  if(pid == master_pid){
144  results.reset(new comm_chunk[k*info.nprocs]);
145  }
146 
147  //this should not be empty!!!
149 
150  //gather at the master process
151  parframepp::rt_base<mpi_par>::gather(proc_results.get(),k,results.get(),master_pid);
152 
153  typename Actor::return_type rslt = Actor::invalid_result();
154 
155  if(pid == master_pid){
156  rslt = actor.get_result(results,k*info.nprocs);
157  }
158 
159  end = std::chrono::system_clock::now();
160  info.runtime = end-start;
161  return std::make_pair(rslt,info);
162  }
