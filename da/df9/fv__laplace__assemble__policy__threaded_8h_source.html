329 void
330 FVLaplaceAssemblyPolicyThreaded<dim, Executor>::AssembleTask<MatrixTp,VectorTp>::apply_boundary_conditions(const std::vector<uint_t>& bfaces ){
331 
332  bool added=false;
333 
334  for(uint_t f=0; f<bfaces.size(); ++f){
335 
336  auto& face = elem_->get_face(bfaces[f]);
337 
338  //get the boundary condition type
339  BCType type = boundary_func_->bc_type(face.boundary_indicator());
340 
341  if(type == BCType::DIRICHLET || type == BCType::ZERO_DIRICHLET)
342  {
343  real_t bc_val = boundary_func_->value(face.centroid());
344  uint_t var_dof = cell_dofs_[0].id;
345  real_t qval = qvals_.empty() ? 1.0 : qvals_[bfaces[f]];
346 
347  //add the boundary condition type
348  b_.add(var_dof, bc_val*fluxes_[bfaces[f]]);
349  auto flux_val = fluxes_[bfaces[f]];
350 
351  //add to the diagonal of the matrix
352  mat_.add_entry(var_dof, var_dof, qval*flux_val);
353 
354  }//Dirichlet
355  else if (type == BCType::NEUMANN) {
356 
357  auto gradient = boundary_func_->gradients(face.centroid());
358  auto normal_comp = dot(gradient, face.normal_vector());
359  uint_t var_dof = cell_dofs_[0].id;
360  b_.add(var_dof, normal_comp);
361  }
362  else if(type == BCType::ZERO_NEUMANN){
363  // nothing to do here.
364  }
365  }
366 }

368 template<int dim, typename Executor>
369 template<typename MatrixTp, typename VectorTp>
370 void
371 FVLaplaceAssemblyPolicyThreaded<dim, Executor>::AssembleTask<MatrixTp,VectorTp>::assemble_one_element(){
372 
373  auto n_dofs = cell_dofs_.size() + neigh_dofs_.size();
374 
375  std::vector<real_t> row_entries(n_dofs, 0.0);
376  std::vector<uint_t> row_dofs(n_dofs);
377 
378  // the indices of the faces local to the
379  // elment that lie on the boundary
380  std::vector<uint_t> boundary_faces;
381  boundary_faces.reserve(elem_->n_faces());
382 
383  //the first dof is the row we work on
384  row_dofs[0] = cell_dofs_[0].id;
385 
386  for(uint_t n=0; n<neigh_dofs_.size(); ++n){
387  row_dofs[n+1] = neigh_dofs_[n].id;
388  }
389 
390  //a dummy index to start counting from 1
391  uint_t dummy_idx =1;
392 
393  for(uint_t f=0; f < elem_->n_faces(); ++f){
394 
395  const auto& face = elem_->get_face(f);
396 
397  if(!face.on_boundary()){
398  //for every face we add to the diagonal entry
399  real_t qval = qvals_.empty() ? 1.0 : qvals_[f];
400 
401  row_entries[0] += qval*fluxes_[f];
402  row_entries[dummy_idx] = -qval*fluxes_[f];
403  dummy_idx++;
404  }
405  else{
406 
407  boundary_faces.push_back(f);
408  }
409  }
410 
411  if(volume_func_){
412  row_entries[0] += volume_func_->value(elem_->centroid())*elem_->volume();
413  }
414 
415  mat_.set_entry(row_dofs[0], row_dofs[0], row_entries[0]);
416  for(uint_t idx=1; idx<row_dofs.size(); ++idx){
417  mat_.set_entry(row_dofs[0], row_dofs[idx], row_entries[idx]);
418  }
419 
420  real_t rhs_val = 0.0;
421  if(rhs_func_ != nullptr){
422  rhs_val = rhs_func_->value(elem_->centroid());
423  }
424 
425  b_.add(cell_dofs_[0].id, rhs_val*elem_->volume());
426 
427  if(!boundary_faces.empty() && boundary_func_ != nullptr){
428  apply_boundary_conditions(boundary_faces);
429  }
430 }
431 
432 }
