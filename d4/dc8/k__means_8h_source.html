118 template<typename Similarity,typename Initializer>
119 typename KMeans<serial,DataInput,CentroidType>::output_type
120 KMeans<serial,DataInput,CentroidType>::cluster(const Similarity& sim,
121  const Initializer& init){
122 
123  std::chrono::time_point<std::chrono::system_clock> start, end;
124  start = std::chrono::system_clock::now();
125 
126  //the object that the algorithm returns
127  KMeansInfo info;
128 
129  size_type ntotal_itrs = 0;
130 
131  random_restart:
132 
133  clusters_.clear();
134  clusters_.resize(this->kmeans_base::data_.k);
135  std::vector<CentroidType> centroids;
136 
137  this->kmeans_base::initialize_clusters(init,*data_in_,centroids,clusters_);
138 
139  const size_type nrows = this->kmeans_base::data_.n_training_examples;
140 
141  for(size_type itr=0; itr<this->kmeans_base::data_.max_n_iterations; ++itr){
142 
143  if(this->kmeans_base::data_.show_iterations){
144  std::cout<<"At iteration: "<<itr<<std::endl;
145  }
146 
147  //loop over the data points
148  for(size_type r=0; r<nrows; /*data_adaptor::size(*data_in_);*/ ++r){
149 
150  CentroidType p = data_adaptor::data_point(*data_in_,r);
151 
152  //cluster the data point
153  kmeans_detail::cluster(p,r,sim,clusters_);
154  }
155 
156  //check for empty cluster
157  bool empty_cluster_found = this->kmeans_base::detect_empty_cluster(clusters_);
158 
159  if(empty_cluster_found &&
160  this->kmeans_base::data_.random_restart_on_empty_cluster){
161 
162  if(this->kmeans_base::data_.show_iterations){
163  std::cout<<"An empty cluster was found restarting..."<<std::endl;
164  }
165 
166  goto random_restart;
167  }
168  else if(empty_cluster_found &&
169  !this->kmeans_base::data_.continue_on_empty_cluster){
170 
171  if(this->kmeans_base::data_.show_iterations){
172  std::cout<<"An empty cluster was found breaking..."<<std::endl;
173  }
174  break;
175  }
176 
177  //recalculate the centroids for each cluster
178  for(size_type c=0; c<clusters_.size(); ++c){
179 
180  clusters_[c].recalculate_centroid(*data_in_);
181  }
182 
183  bool converged = this->kmeans_base::detect_convergence(sim,clusters_,centroids);
184 
185  if(converged){
186  info.converged = converged;
187 
188  if(this->kmeans_base::data_.show_iterations){
189  std::cout<<"KMeans converged with tolerance: "<<this->kmeans_base::data_.tolerance<<std::endl;
190  }
191 
192  break;
193  }
194 
195  for(size_type c=0; c<centroids.size(); ++c){
196  centroids[c] = clusters_[c].location;
197  }
198 
199  ntotal_itrs++;
200  }
201 
202  info.clusters.resize(clusters_.size());
203 
204 
205  for(size_type c=0; c<clusters_.size(); ++c){
206  info.clusters[c]=std::make_pair(c,clusters_[c].points.size());
207  }
208  info.niterations = ntotal_itrs;
209  end = std::chrono::system_clock::now();
210  info.runtime = end-start;
211 
212  return info;
213 }

370 template<typename DataInput,typename CentroidType>
371 template<typename Similarity,typename Initializer>
372 typename KMeans<cput_c11,DataInput,CentroidType>::output_type
373 KMeans<cput_c11,DataInput,CentroidType>::cluster(const Similarity& sim,
374  const Initializer& init){
375 
376 #ifdef PARML_DEBUG
377  using utilities::ExeEmptyPartitions;
378  const std::string msg="The partitions have not been assigned";
379  Assert(this->has_partitions()==true,ExeEmptyPartitions(msg));
380 #endif
381 
382  std::chrono::time_point<std::chrono::system_clock> start, end;
383  start = std::chrono::system_clock::now();
384 
385  //create the tasks
386  size_type nthreads = ParML::n_threads();
387 
388  //the object that the algorithm returns
389  KMeansInfo info;
390  info.nthreads = nthreads;
391 
392  size_type ntotal_itrs = 0;
393 
394  random_restart:
395 
396  clusters_.clear();
397  clusters_.resize(this->kmeans_base::data_.k);
398  std::vector<CentroidType> centroids;
399 
400  //initialize the clusters
401  this->kmeans_base::initialize_clusters(init,*data_in_,centroids,clusters_);
402 
403  if(remove_task_== nullptr){
404 
405  remove_task_ = new par_kmeans_detail::RemoveTask<CentroidType>(clusters_);
406  }
407 
408  size_type nparts = this->n_partitions();
409 
410  //resize the number of tasks
411  tasks_.resize(nparts-1,nullptr);
412 
413  //allocate space for tasks. Keep the last task for this thread
414  for(size_type t=0; t<nparts-1; ++t){
415 
416  tasks_[t] = new par_kmeans_detail::Task<DataInput,CentroidType,Similarity>(this->get_partition(t),*this->data_in_,sim);
417 
418  //for every task created assign the centroids for its clusters
419  std::vector<kmeans_detail::Cluster<CentroidType> >& task_clusters =
420  static_cast<par_kmeans_detail::Task<DataInput,CentroidType,Similarity>*>(tasks_[t])->get_local_clusters();
421 
422  task_clusters.resize(this->kmeans_base::data_.k);
423 
424  for(size_type c=0; c<centroids.size(); ++c){
425 
426  task_clusters[c].id = c;
427  task_clusters[c].location = centroids[c];
428  }
429  }
430 
431  for(size_type itr=0; itr<this->kmeans_base::data_.max_n_iterations; ++itr){
432 
433  if(this->kmeans_base::data_.show_iterations){
434  std::cout<<"At iteration: "<<itr<<std::endl;
435  }
436 
437  //schedule the tasks
438  parframepp::rt_cput::schedule_tasks_async(tasks_);
439 
440  const utilities::range1d<size_type>& part = this->get_partition(nparts-1);
441 
442  size_type local_begin = part.begin(); //partitions_[nparts-1].begin();
443  size_type local_end = part.end(); //partitions_[nparts-1].end();
444 
445  //loop over the data points assigned for this thread
446  //this is the main thread
447  for(size_type r=local_begin; r<local_end; ++r){
448 
449  CentroidType p = data_adaptor::data_point(*data_in_,r);
450 
451  //cluster the data point
452  kmeans_detail::cluster(p,r,sim,clusters_);
453  }
454 
455 
456  //wait for the other threads to finish and collect the
457  //local results;
458  parframepp::task_barrier::barrier(tasks_);
459 
460  for(size_type t=0; t<tasks_.size(); ++t){
461 
462  std::vector<kmeans_detail::Cluster<CentroidType> >& task_clusters =
463  static_cast<par_kmeans_detail::Task<DataInput,CentroidType,Similarity>*>(tasks_[t])->get_local_clusters();
464  //get the clusters
465  merge_clusters(task_clusters);
466  }
467 
468  //check for empty cluster
469  bool empty_cluster_found = this->kmeans_base::detect_empty_cluster(clusters_);
470 
471  if(empty_cluster_found && this->kmeans_base::data_.random_restart_on_empty_cluster){
472  if(this->kmeans_base::data_.show_iterations){
473  std::cout<<"An empty cluster was found restarting..."<<std::endl;
474  }
475  goto random_restart;
476  }
477  else if(empty_cluster_found && !this->kmeans_base::data_.continue_on_empty_cluster){
478  if(this->kmeans_base::data_.show_iterations){
479  std::cout<<"An empty cluster was found breaking..."<<std::endl;
480  }
481  break;
482  }
483 
484  //recalculate the centroids for each cluster
485  for(size_type c=0; c<clusters_.size(); ++c){
486  clusters_[c].recalculate_centroid(*data_in_);
487  }
488 
489  bool converged = this->kmeans_base::detect_convergence(sim,clusters_,centroids);
490 
491  if(converged){
492  info.converged = converged;
493  if(this->kmeans_base::data_.show_iterations){
494  std::cout<<"Kmeans converged with tolerance: "<<this->kmeans_base::data_.tolerance<<std::endl;
495  }
496 
497  break;
498  }
499 
500  for(size_type c=0; c<centroids.size(); ++c){
501  centroids[c] = clusters_[c].location;
502  }
503 
504  //reset centroids for the tasks
505  for(size_type t=0; t<tasks_.size(); ++t){
506 
507  std::vector<kmeans_detail::Cluster<CentroidType> >& task_clusters =
508  static_cast<par_kmeans_detail::Task<DataInput,CentroidType,Similarity>*>(tasks_[t])->get_local_clusters();
509 
510  for(size_type c=0; c<centroids.size(); ++c){
511  task_clusters[c].location = centroids[c];
512  task_clusters[c].points.clear();
513  }
514 
515  //set the task state as pending
516  tasks_[t]->set_state(parframepp::TaskBase::TaskState::PENDING);
517  }
518 
519  ntotal_itrs++;
520  }
521 
522  info.clusters.resize(clusters_.size());
523 
524 
525  for(size_type c=0; c<clusters_.size(); ++c){
526  info.clusters[c]=std::make_pair(c,clusters_[c].points.size());
527  }
528  info.niterations = ntotal_itrs;
529  end = std::chrono::system_clock::now();
530  info.runtime = end-start;
531 
532  return info;
533 }
