99 KMeans<ClusterType>::cluster(const DataIn& data, const Similarity& similarity, const Initializer& init){

101 
102  typedef typename KMeans<ClusterType>::cluster_t cluster_t;
103  typedef typename KMeans<ClusterType>::output_t output_t;
104 
105  output_t info;
106  auto k = control_.k;
107  auto rows = data.rows();
108 
109 
110  // more clusters than data does not make
111  // sense
112  if(k > rows){
113  throw std::logic_error("Number of clusters cannot be larger than number of rows");
114  }
115 
116  //start timing
117  std::chrono::time_point<std::chrono::system_clock> start, end;
118  start = std::chrono::system_clock::now();
119 
120  random_restart:
121 
122  //the old centroids
123  std::vector<typename ClusterType::point_t> centroids;
124 
125  {
126  std::vector<cluster_t> empty;
127  clusters_.swap(empty);
128  }
129 
130  //initialize the clusters
131  init(data, k, centroids);
132 
133  if(centroids.size() != k){
134  throw std::logic_error("Incorrect centroid initialization: "+
135  std::to_string(centroids.size()) +
136  " not equal to: "+
137  std::to_string(k));
138  }
139 
140 
141  clusters_.resize(k);
142  for(uint_t c=0; c<clusters_.size(); ++c){
143  clusters_[c].centroid = centroids[c];
144  clusters_[c].id = c;
145  }
146 
147  bool converged = false;
148  real_t residual = std::numeric_limits<real_t>::max();
149  bool empty_cluster = false;
150  uint_t empty_cluster_id = kernel::KernelConsts::invalid_size_type();
151 
152  while (control_.continue_iterations()) {
153 
154  if(control_.show_iterations()){
155 
156  std::cout<<"\tK-means iteration: "<<control_.get_current_iteration()<<std::endl;
157  }
158 
159  for(uint_t r=0; r<data.rows(); ++r){
160 
161  // try to cluster the point
162  cluster_point_(kernel::get_row(data, r), r, similarity);
163  }
164 
165  // check if we have empty clusters
166  auto [empty_cluster, empty_cluster_id] = check_empty_clusters_();
167 
168  if(empty_cluster){
169 
170  if ( control_.show_iterations()){
171  std::cout<<"\t\tEmpty cluster detected..."<<std::endl;
172  }
173 
174  if(!control_.continue_on_empty_cluster && control_.random_restart_on_empty_cluster){
175 
176  if(control_.show_iterations()){
177  std::cout<<"\t\tRestarting..."<<std::endl;
178  }
179 
180  goto random_restart;
181  }
182  else if(!control_.continue_on_empty_cluster){
183  break;
184  }
185  else if(control_.show_iterations()){
186  std::cout<<"\t\tContinue with empty cluster detected..."<<std::endl;
187  }
188  }
189 
190  // clalculate new centroids
191  calculate_new_centroids_(data);
192 
193  // check if we converged
194  auto [converged, residual] = detect_convergence_(similarity, centroids);
195 
196  control_.update_residual(residual);
197 
198  // update the centroids also
199  for(uint_t c=0; c<clusters_.size(); ++c){
200  centroids[c] = clusters_[c].centroid;
201  }
202 
203  if(control_.show_iterations()){
204 
205  std::cout<<"\t\t Residual at teration: "<<residual<<std::endl;
206  }
207  }
208 
209  auto state = control_.get_state();
210  end = std::chrono::system_clock::now();
211 
212  info.runtime = end-start;
213  info.nprocs = 1;
214  info.nthreads = 1;
215  info.converged = state.converged;
216  info.residual = state.residual;
217  info.tolerance = state.tolerance;
218  info.niterations = state.num_iterations;
219  return info;

224 std::tuple<bool, real_t>
225 KMeans<ClusterType>::detect_convergence_(const Similarity& sim,
226  const std::vector<typename ClusterType::point_t>& old_centers)const{
227 
228  if(clusters_.size() != old_centers.size()){
229  const std::string msg = "Current number of clusters: "+std::to_string(clusters_.size())+
230  " not equal of old number of clusters: "+std::to_string(old_centers.size());
231  throw std::logic_error(msg);
232  }
233 
234  bool converged = true;
235  auto dis = std::numeric_limits<real_t>::max();
236 
237  for(auto c=0; c<clusters_.size(); ++c){
238 
239  dis = sim(clusters_[c].centroid, old_centers[c]);
240 
241  if(dis > Similarity::tolerance()){
242  converged = false;
243  break; //we don't have to continue calculating distances
244  }
245  }
246 
247  return {converged, dis};

252 void
253 KMeans<ClusterType>::cluster_point_(const point_t& point, uint_t pid, const Similarity& sim ){
254 
255  typedef typename Similarity::value_t value_t;
256 
257  value_t current_dis = std::numeric_limits<value_t>::max();
258  auto cluster_id = kernel::KernelConsts::invalid_size_type();
259 
260  for(auto c=0; c<clusters_.size(); ++c){
261 
262  auto dis = sim(point, clusters_[c].centroid);
263 
264  if(dis - current_dis < Similarity::tolerance()){
265 
266  current_dis = dis;
267  cluster_id = c;
268  }
269  }
270 
271  if(cluster_id >= clusters_.size()){
272 
273  //make sure we have a valid cluster id
274  const std::string msg = "Cluster id: "+
275  std::to_string(cluster_id)+" not in [0,"+
276  std::to_string(clusters_.size())+")";
277  throw std::logic_error(msg);
278  }
279 
280 
281  // add to the new cluster
282  // and remove the id from the others. Since each point is assigned
283  // to exactly one cluster we don't have to remove from the rest
284  // of the clusters if the point was found.
285 
286  bool found = clusters_[cluster_id].add_to_cluster(pid);
287 
288  if(!found){
289 
290  for(auto c=0; c<clusters_.size(); ++c){
291 
292  if(c!= cluster_id && !found){
293  found = clusters_[c].remove_from_cluster(pid);
294  }
295  }
296  }
297 }
