224  //nothing to do if this cluster has no

226  if(points.empty()){
227  return;

229 
230  size_type r = location.rows();
231  size_type c = location.cols();
232 
233  //the new centroid will be the average

235  DataPoint avg(r,c);
236 
237  for(size_type p=0; p<points.size(); ++p){
238

240  }
241 
242  avg /= points.size();
243 
244  location = avg;

296  template<typename DataPoint,typename Similarity>
297  void cluster(const DataPoint& point,size_type pid,
298  const Similarity& sim, std::vector<Cluster<DataPoint> >& clusters){
299 
300  typedef typename Similarity::value_type value_type;
301 
302  value_type current_dis = std::numeric_limits<value_type>::max();
303  size_type cluster_id = ParML::iuint();
304 
305  for(size_type c=0; c<clusters.size(); ++c){
306 
307  value_type dis = sim(point,clusters[c].location);
308 
309  if(dis -current_dis < ParML::tolerance()){
310 
311  current_dis = dis;
312  cluster_id = c;
313  }
314  }
315 
316 #ifdef PARML_DEBUG
317 
318  //make sure we have a valid cluster id
319  const std::string msg = "Cluster id: "+
320  std::to_string(cluster_id)+" not in [0,"+
321  std::to_string(clusters.size())+")";
322  Assert(cluster_id < clusters.size(),ExeInvalidClusterId(msg));
323 #endif
324 
325  //add to the new cluster
326  //and remove the id from the others. Since each point is assigned
327  //to exactly one cluster we don't have to remove from the rest
328  //of the clusters if the point was found.
329 
330  bool found = clusters[cluster_id].add_to_cluster(pid);
331 
332  if(!found){
333 
334  for(size_type c=0; c<clusters.size(); ++c){
335 
336  if(c!= cluster_id && !found){
337  found = clusters[c].remove_from_cluster(pid);
338  }
339  }
340  }
341  }
