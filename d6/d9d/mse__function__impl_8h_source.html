158 template<typename HypothesisFn, typename DataSetType,
159  typename LabelsType, typename RegularizerFn>
160 typename MSEFunction<SigmoidFunction<HypothesisFn>, DataSetType,
161 LabelsType, RegularizerFn>::output_t
162 MSEFunction<SigmoidFunction<HypothesisFn>, DataSetType,
163  LabelsType, RegularizerFn>::value(const DataSetType& dataset, const LabelsType& labels)const{
164 
165 
166  if(dataset.rows() != labels.size()){
167  throw std::invalid_argument("Invalid number of data points and labels vector size");
168  }
169 
170  typedef typename MSEFunction<SigmoidFunction<HypothesisFn>,
171  DataSetType, LabelsType, RegularizerFn>::output_t output_t;
172  output_t result = output_t();
173 
174  for(uint_t row_idx=0; row_idx<dataset.rows(); ++row_idx){
175 
176  auto row = get_row(dataset, row_idx);
177  auto y = labels[row_idx];
178  auto hypothesis_value = this->h_ptr_->value(row);
179 
180  //h is close to one
181  if(std::fabs(1 - hypothesis_value) < KernelConsts::tolerance()){
182 
183  //we plug a large error contribution if y is anything than one
184  if( y != 1){
185  result += 1.0;
186  }
187  }
188  else if(std::fabs(hypothesis_value) < KernelConsts::tolerance()){
189 
190  //hval is zero. we only get contribution
191  //if the label is not zero as well
192  if( y > KernelConsts::tolerance()){
193  result += 1.0;
194  }
195  }
196  else{
197  //do it normally
198  //calculate the logarithms and check if they are
199  //infinite or nana
200  auto log_one_minus_h = std::log(1 - hypothesis_value);
201  auto log_h = std::log(hypothesis_value);
202  result += y*log_h +(1 - y)*log_one_minus_h;
203  }
204  }
205 
206  result *= -1;
207  result /= dataset.rows();
208 
209  // this is a valid result
210  result.validate_result();
211  return result;
212 }

446 MSEFunction<HypothesisFn, PartitionedType<DataSetType>,

452  throw std::invalid_argument("Invalid number of data points and labels vector size");
453  }
454 
455  if(value_tasks_.empty()){
456  execute_value_tasks_(dataset, labels, executor, options);
457  }
458  else{
459 
460  for(uint_t t=0; t<executor.get_n_threads(); ++t){
461  value_tasks_[t]->reschedule();
462  }
463 
465  executor.execute(value_tasks_, options);
466  }
467 
468  // in any case now we have tasks that should have finished
469  // by default we assume the result is valid

471 
472  for(uint_t t=0; t < value_tasks_.size(); ++t){
473 
474  // if we reached here but for some reason the
475  // task has not finished properly invalidate the result

477  result.invalidate_result(false);
478  }
479  else{
480 
481  result += value_tasks_[t]->get_result();
482  }
483  }
484 
485  if(!result.is_result_valid()){
486  return result;
487  }
488 
489  result /= dataset.rows();
490 
491  if(this->r_ptr_){
492  result += this->r_ptr_->value(executor, options, dataset, labels );
493  }

667  }
668  else if(std::fabs(hypothesis_value) < KernelConsts::tolerance()){

673  result += 1.0;
674  }
675  }
676  else{
677  //do it normally
678  //calculate the logarithms and check if they are
679  //infinite or nana

681  auto log_h = std::log(hypothesis_value);
682  result += y*log_h +(1 - y)*log_one_minus_h;
683  }
684  }
