193 template<typename PolicyTp, typename DynamicsP>
194 void
195 SyncValueFuncItr<WorldTp>::step(PolicyTp& policy,
196  const DynamicsP& dynamics){
197  real_t delta = 0.0;
198 
199  // loop over the states of the world
200  for(uint_t s=0; s<world_->n_states(); ++s){
201 
202  // get the s-th state
203  auto state = world_->get_state(s);
204 
205  // the world should know which state is terminal
206  if(!world_->is_goal_state(state)){
207 
208  // this is not the goal state
209  real_t old_v = vold_[state.get_id()];
210 
211  real_t weighted_sum = 0.0;
212 
213  // loop over all the actions allowed on this
214  // state
215  for(uint_t a=0; a<state.n_actions(); ++a){
216 
217  auto action = state.get_action(a);
218 
219  // get the probability at this state
220  // to take the given action
221  auto action_prob = policy(action, state);
222 
223  // loop over the states we can transition
224  // to from this state
225  auto transition_states = state.get_states();
226 
227  auto value = 0.0;
228  for(uint_t os=0; os < transition_states.size(); ++os){
229 
230  if(transition_states[os]){
231 
232  // the reward we will receive if at the
233  // current state we take the given action.
234  // This means that the agent transitions to
235  // transition_states[os]
236  real_t r = world_->get_reward(state, action);
237  real_t vs_prime = vold_[transition_states[os]->get_id()];
238  auto p= dynamics(*transition_states[os], r, state, action);
239  value += p*(r + imput_.gamma*vs_prime );
240  }
241  }
242 
243  weighted_sum += action_prob*value;
244  }
245 
246  v_[state.get_id()] = weighted_sum;
247  delta = std::max(delta, std::fabs(old_v-weighted_sum));
248  }
249 }
250 
251  itr_controller_.update_residual(delta);
252 
253  // finally update the vectors
254  vold_ = v_;
255 }
