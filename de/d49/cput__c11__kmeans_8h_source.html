403 template<typename DataIn,typename Similarity,typename Initializer>
404 void
405 KMeans<parframepp::platform_type::Type::CPUT_C11,DataPoint>::cluster(const DataIn& data,
406  const Similarity& similarity,const Initializer& init){
407 
408  size_type k = this->cntrl_.k;
409  size_type rows = data.n_rows();
410  size_type max_itrs = this->cntrl_.max_n_iterations;
411 
412 #ifdef PARML_DEBUG
413  const std::string func_msg = "In KMeans<parframepp::platform_type::Type::CPUT_C11,DataPoint>::cluster(). ";
414  kmeans_detail::assert_comon(func_msg,rows,k,max_itrs);
415 
416  //make sure that we have partitions
417  std::string msg = func_msg+" No partitions have been set.";
418  using utilities::ExeLogicError;
419  Assert(this->has_partitions(),ExeLogicError(msg));
420 #endif
421 
422  this->info_.nthreads = parframepp::rt_cput::n_threads();
423  this->info_.n_clustering_points = data.n_rows();
424  this->info_.residual = this->cntrl_.tolerance;
425 
426  //start timing
427  std::chrono::time_point<std::chrono::system_clock> start, end;
428  start = std::chrono::system_clock::now();
429 
430  random_restart:
431 
432 
433  //the old centroids
434  std::vector<DataPoint> centroids;
435 
436  //initialize the clusters
437  kmeans_detail::DataOnProcInitializer::initialize(data,init,this->cntrl_.k,this->clusters_,centroids);
438 
439  //the exit condition
440  kmeans_detail::ExitCondition exit_cond(this->cntrl_.max_n_iterations);
441 
442  //the object that handles the iteration
443  kmeans_detail::Iteration<parframepp::platform_type::Type::CPUT_C11> iteration(this->cntrl_);
444 
445  //create the tasks
446  iteration.create_tasks(data,similarity,this->get_partitions());
447 
448  //what the algorithm should do at each iteration
449  typename KMeansBase<DataPoint>::ActionType action = KMeansBase<DataPoint>::ActionType::INVALID_TYPE;
450 
451  while(exit_cond.check() == false){
452 
453  if(this->cntrl_.show_iterations){
454 
455  std::cout<<"\tK-means iteration: "<<exit_cond.current_itr<<std::endl;
456  }
457 
458  action = iteration(data,similarity,this->clusters_,centroids);
459 
460  if(action == KMeansBase<DataPoint>::ActionType::RESTART)
461  goto random_restart;
462  else if(action == KMeansBase<DataPoint>::ActionType::BREAK)
463  break;
464  else if(action == KMeansBase<DataPoint>::ActionType::CONVERGED){
465  this->info_.converged = true;
466  break;
467  }
468  }
469 
470  if(action == KMeansBase<DataPoint>::ActionType::CONVERGED){
471 
472  this->info_.clusters.resize(this->clusters_.size());
473  for(size_type c=0; c<this->clusters_.size(); ++c){
474  this->info_.clusters[c]=std::make_pair(c,this->clusters_[c].points.size());
475  }
476  }
477  this->info_.niterations = exit_cond.current_itr;
478  end = std::chrono::system_clock::now();
479  this->info_.runtime = end-start;
480 }
