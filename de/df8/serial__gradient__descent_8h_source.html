126

129

131  coeffs[c] -= input_.learning_rate*j_grads[c];
132  }
133 
134  // reset again the coeffs
135  h.set_coeffs(coeffs);
136 
137  //recalculate...
138  j_current = err_function_.value(data, y).get_resource();
139 
140  real_t error = std::fabs(j_current - j_old);
141  input_.update_residual(error);
142  uint_t itr = input_.get_current_iteration();
143
