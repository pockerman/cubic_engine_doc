342 void
343 knn_policy_base<is_regressor>::operator()(const DataMat& data, const LabelType& labels, const DataVec& point,
344  const Similarity& sim, const kernel::range1d<uint_t>& range){
345 
346  typedef typename knn_policy_base<is_regressor>::Pair Pair;
347 
348  //we will calculating as many distances as
349  //the size of the range
350  std::vector<Pair> distances;
351  distances.reserve(range.size());
352 
353  for(uint_t r=range.begin(); r<range.end(); ++r){
354 
355  //access the r-th row of the matrix
356  const DataVec& data_point = kernel::get_row(data, r);
357 
358  //compute the distance between the data point and the
359  //input point
360  real_t dis = sim(data_point, point);
361 
362  //compute the similarity and append it to distances
363  distances.push_back(std::make_pair(r,dis));
364  }
365 
366  //sort the computed pairs based on the distances
367  std::sort(distances.begin(),distances.end(),
368  [](const Pair& p1,const Pair& p2){return p1.second < p2.second;});
369 
370  {
371  //empty what has been already computed
372  std::vector<Pair> empty;
373  data_handler_.k_distances.swap(empty);
374  data_handler_.majority_vote.clear();
375  }
376 
377  data_handler_.k_distances.reserve(data_handler_.k);
378 
379 
380  for(uint_t i=0; i<data_handler_.k; ++i){
381  data_handler_.k_distances.push_back(distances[i]);
382  }
383 
384  //fill in the majority vote
385  fillin_majority_vote(labels);
